import numpy as np
x1=[[1.2,4.2,1.9,1.0],
   [5.7,6.4,9.0,1.5],
   [2.4,6.5,2.9,9.9]]

class Dense(): 
  def __init__(self,inputs,neurons):
    self.weights=0.10*(np.random.randn(inputs,neurons))
    self.bias=np.zeros((1,neurons))
  def forward(self,inputs):
    self.output=np.dot(inputs,self.weights)+self.bias

layer1=Dense(4,5)

layer1

layer1.forward(x1)

print(layer1.output)

layer11=Dense(5,2)

layer11

layer11.forward(layer1.output)

print(layer11.output)

class Relu:
  def forward(self,inputs):
    self.output=np.maximum(0,inputs)

activation= Relu()
activation.forward(layer1.output)

print(activation.output)

layer11.forward(activation.output)

print(layer11.output)

activation.forward(layer11.output)

print(activation.output)
